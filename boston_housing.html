def performance_metric2(y_true, y_predict):
    """"""
    y_true_mean = float(sum(y_true))/len(y_true)
    
    #y_true
    square_error_line = sum([(a-b)*(a-b) for a,b in zip(y_true, y_predict)])
    #y_true
    square_error_mid = sum([(a-y_true_mean)*(a-y_true_mean) for a in y_true])
    #R2_score
    score = 1- square_error_line/square_error_mid

    return score

    """  [X,y]"""
    
    cross_validator = KFold(10)

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    len_valid = len(X)/cv
    X = np.array(X)
    y = np.array(y)
    
    scores = []
    for i in range(cv):
        valid_index = range(i*len_valid, i*len_valid+len_valid)
        x_valid = X[valid_index]
        x_train = np.delete(X, valid_index, axis = 0)
        y_valid = y[valid_index]
        y_train = np.delete(y, valid_index)
        
        estimator.fit(x_train, y_train)
        y_predict = estimator.predict(x_valid)
        s = performance_metric(y_valid, y_predict)
        scores.append(s)

    return scores


def fit_model2(X, y):
    """  [X,y]"""
    cv = 10
    range_arr = range(1,11)
    reg_arr = [DecisionTreeRegressor(max_depth = depth) for depth in range_arr]
    score_arr = []
    for reg in reg_arr:
        scores = cross_val_score(reg, X, y, scoring=performance_metric, cv=cv)
        score_arr.append(np.mean(scores))
    
    #
    max_index = np.argmax(score_arr)
    best_estimator = reg_arr[max_index]
    
    return best_estimator
